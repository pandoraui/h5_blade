单独使用 

demo01: function () {
    if (!this.tab01) {
        this.tab01 = new UITab({
            wrapper: $('.tab1-wrap'),
            data: [{
                id:1,
                name: '好评'
            },
            {
                id:2,
                name: '中评'
            },
            {
                id:3,
                name: '差评'
            }]

        });

    }

    this.tab01.show();
},
自定义使用 

demo02: function () {
    if (!this.tab02) {
        this.tab02 = new UITab({
            wrapper: $('.tab2-wrap'),
            data: [{
                id:1,
                name: '好评'
            },
                {
                    id:2,
                    name: '中评'
                },
                {
                    id:3,
                    name: '差评'
                }],

            onChange: function() {
                console.log('onChange callback');
            }


        });

    }

    this.tab02.show();
}
